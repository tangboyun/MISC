#+begin_src latex
  \begin{block}{评价准则}
     \begin{itemize}
     \item 较好的预后预测 (Good prognostic value / prediction accuracy)
     \item 稳定的生物学解读 (Functional similarity / Semantic similarity)
     \end{itemize}
  \end{block}
  \pause
  \begin{block}{难点与挑战}
     \begin{itemize}
     \item 属于高维拟合问题 $p \gg N $
     \item Response值多为截尾数据
     \end{itemize}
  \end{block}
#+end_src
